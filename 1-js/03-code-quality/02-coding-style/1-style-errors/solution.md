A variante corrigida:

```js
function pow(x, n) {
  let result = 1;

  for (let i = 0; i < n; i++) {
    result *= x;
  }

  return result;
}

let x = prompt("x?", "");
let n = prompt("n?", "");

if (n < 0) {
  alert(`A potência de ${n} não é suportada, por favor insira um número inteiro maior do que zero`);
} else {
  alert( pow(x, n) );
}
```
